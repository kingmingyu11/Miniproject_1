/*테이블 삭제*/
drop table 소비자_주문_테이블;

/*혹시 모를 겹침을 방지하기 위해서 Battery Recycle 의 BR을 붙였습니다.*/
drop table BRORDER;
drop table BRCOMPOUND;
drop table BRFIRM;
drop table BRMONTH;
drop table BRDAY;

/*테이블 생성*/
create table BRORDER(
주문번호 number(20) primary key,
날짜 date,
업체종류 varchar(20),
화합물이름 varchar(20),
무게 number(20),
납부여부 varchar(20));

CREATE TABLE BRCOMPOUND(
    화합물_ID NUMBER(2) PRIMARY KEY,
    화합물이름 VARCHAR2(50),
    무게 NUMBER(10)
);

create table BRFIRM(
ID NUMBER(20) PRIMARY KEY,
업체명 VARCHAR(20),
폐배터리개수 NUMBER(20)
);

CREATE TABLE BRMONTH(
    날짜 NUMBER(2) PRIMARY KEY,
    총량 NUMBER(30)
);

CREATE TABLE BRDAY(
화합물_ID NUMBER(20) PRIMARY KEY,
화합물이름 VARCHAR2(50),
무게 NUMBER(10));
